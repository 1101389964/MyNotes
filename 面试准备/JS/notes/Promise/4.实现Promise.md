## Promise.resolve

```js
Promise.myResolve = function (value) {
    // 是Promise实例，直接返回即可
    if (value && typeof value === 'object' && (value instanceof Promise)) {
      return value
    }
    // 否则其他情况一律再通过Promise包装一下 
    return new Promise((resolve) => {
      resolve(value)
    })
  }
```

## Promise.reject

```js
Promise.myReject = function (value) {
  return new Promise((_, reject) => {
    reject(value)
  })
}
```

## Promise.all

```js
// bad
Promise.myall = function(agrs) {
  if (agrs.length === 0) {
    return Promise.resolve([]);
  }
  const result = [];
  for (let i = 0; i < agrs.length; i++) {
    // 有的数组项有可能不是Promise，需要手动转化一下
    Promise.resolve(agrs[i]).then(res => {
      // 这里不能使用push加入pormise，如果摸个pomise使用定时器延时，传入的promise不会顺序插入result
      result.push(res)
    }).catch(err => {
      // 这里的return不会触发函数的返回，属于是catch内部的返回
      return Promise.reject(err);
    })
  }
  return Promise.resolve(result);
}

// good
Promise.myall = (promises) => {
  // 直接在pormise内部处理，当可以遍历完毕就reslove，碰到catch则resolve
  return new Promise((resolve, reject) => {
    if (promises.length === 0) {
      return reslove([])
    }
    const result = [];
    for (let i = 0; i < promises.length; i++) {
      // 有的数组项有可能不是Promise，需要手动转化一下
      Promise.resolve(promises[i]).then(res => {
        result[i] = res;
        if (i === promises.length - 1) resolve(result); 
      }).catch(res => {
        reject(res)
      }) 
    }
  })
}
```

## Promise.myallSettled

```js
Promise.myallSettled = (promises) => {
  return new Promise((resolve, reject) => {
    if (promises.length === 0) {
      return reslove([])
    }
    const result = [];
    for (let i = 0; i < promises.length; i++) {
      Promise.resolve(promises[i]).then(res => {
        result[i] = {status: 'fulfilled', value: res};
      }).catch(res => {
        result[i] = {status: 'rejected', value: res};
      })
    }
    return resolve(result)
  })
}
```

## Promise.race

```js
Promise.myRace = (promises) => {
  return new Promise((resolve, reject) => {
    promises.forEach(item => {
      item.then(res => resolve(res)).catch(err => reject(err))
    });
  })
}
```
