## Promise.reject

```js
Promise.myReject = function (value) {
  return new Promise((_, reject) => {
    reject(value)
  })
}
```

## Promise.race

```js
Promise.myRace = (promises) => {
  return new Promise((resolve, reject) => {
    promises.forEach(item => {
      item.then(res => resolve(res)).catch(err => reject(err))
    });
  })
}
```
