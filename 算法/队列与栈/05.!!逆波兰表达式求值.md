  var evalRPN = function(tokens) {
      const map = {
          '+' : (b, a) => a + b,
          '-' : (b, a) => a - b,
          '*' : (b, a) => (a * b) >> 0,
          '/' : (b, a) => (a / b) >> 0,
      }
      const stack = [];
      for (let i = 0; i < tokens.length; i++) {
          if (map.hasOwnProperty(tokens[i])) {
              const res = map[tokens[i]](stack.pop(), stack.pop());
              stack.push(res);
          }else {
              stack.push(Number(tokens[i]));
          }
      }
      return stack.pop();
  };
